################################################################################

#   xclip

################################################################################
#	VARIABLES HERE
################################################################################
FILE="$1"
OUT="$(echo $FILE | grep -o "^[a-zA-Z][^.]*")"'.html'
IFS='\
'

################################################################################
#	FUNCTIONS HERE
################################################################################
function sedHTML() {
  sed -iE 's/&/\&amp\;/g;s/</\&lt\;/g;s/>/\&gt\;/g;s/"/\&quot\;/g' "$OUT"
}

################################################################################
#	BEGINNING OF MAIN
################################################################################
cp $FILE $OUT
sedHTML
sed -iE '1s/^/<pre id="code">\
/' "$OUT"
printf '</pre>' >> $OUT
cat $OUT | xclip -selection c
#rm $OUT
