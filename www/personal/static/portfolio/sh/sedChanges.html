# Purpose: bash script for rewriting python scripts 
#         
################################################################################
#	VARIABLES HERE
################################################################################
FILE=$1
SUBE=$2
SUBS=$3

################################################################################
#	FUNCTIONS HERE
################################################################################
function sedChanges() {
  sed -i '' "s/\(self.choice = \"\)[a-z]/\1${SUBE}/" $FILE
  sed -i '' "s/\(self.script = \"\)[a-zA-Z_]*/\1${SUBS}/" $FILE
}

################################################################################
#	BEGINNING OF MAIN
################################################################################
sedChanges
