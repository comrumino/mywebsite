#define WIN32_LEAN_AND_MEAN
#include "stdafx.h"
#include <windows.h>
#include <winbase.h>
#include <tchar.h>
#include <string>

//Fundamental audience algorithm solution
int _tmain(int argc, _TCHAR* argv[]){
  int p=0,a=0,f=0;
  std::string str = "10101024";
  for (char& c : str) {
    a = a + (int)c - 48; 
    f = a+f < p ? p-a : f;
    p++;  
  }
  printf("%i,%i,%i\n", p, a, f);
  getchar();
}
