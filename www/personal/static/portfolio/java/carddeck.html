import java.util.*;
public class CardDeck {

    private static final Map<String, String> PrettifyMap;
    static {
        PrettifyMap = new HashMap<>();
        PrettifyMap.put("TWO", "2");
        PrettifyMap.put("THREE", "3");
        PrettifyMap.put("FOUR", "4");
        PrettifyMap.put("FIVE", "5");
        PrettifyMap.put("SIX", "6");
        PrettifyMap.put("SEVEN", "7");
        PrettifyMap.put("EIGHT", "8");
        PrettifyMap.put("NINE", "9");
        PrettifyMap.put("TEN", "10");
        PrettifyMap.put("JACK", "Jack");
        PrettifyMap.put("QUEEN", "Queen");
        PrettifyMap.put("KING", "King");
        PrettifyMap.put("ACE", "Ace");
        PrettifyMap.put("HEART", "Hearts");
        PrettifyMap.put("SPADE", "Spades");
        PrettifyMap.put("CLUB", "Clubs");
        PrettifyMap.put("DIAMOND", "Diamonds");
    }

    public static void main (String[] args) {
        int count = 0;
        ArrayList<Card> fiftyTwoCards = new ArrayList<Card>();
        for (int i = 0; i < 13; i++) {
            FaceValue faceValue = FaceValue.values()[i];
            for (int j = 0; j < 4; j++) {
                count += 1;
                Card card = new Card( faceValue, Suit.values()[j] );
                fiftyTwoCards.add(card);
            }
        }

        Collections.shuffle(fiftyTwoCards);
        for (Iterator<Card> cardsIter = fiftyTwoCards.iterator(); cardsIter.hasNext(); ) {
            System.out.format("Cards remaining: %d\n", count);
            count -= 1;
            Card currentCard = cardsIter.next();
            String faceValue = PrettifyMap.get(currentCard.getFaceValue().toString());
            String suit = PrettifyMap.get(currentCard.getSuit().toString());
            System.out.println( faceValue + " of " + suit );
        }
    }
}

class Card {
    private Suit suit;
    private FaceValue faceValue;

    public Card( FaceValue faceValue, Suit suit) {
        this.faceValue = faceValue;
        this.suit = suit;
    }
    public Suit getSuit() {
        return suit;
    }
    public FaceValue getFaceValue() {
        return faceValue;
    }
    public void setSuit(Suit suit) {
        this.suit = suit;
    }
    public void setFaceValue(FaceValue faceValue) {
       this.faceValue = faceValue;
    }
}

enum FaceValue {
    TWO("TWO"),
    THREE("THREE"),
    FOUR("FOUR"),
    FIVE("FIVE"),
    SIX("SIX"),
    SEVEN("SEVEN"),
    EIGHT("EIGHT"),
    NINE("NINE"),
    TEN("TEN"),
    JACK("JACK"),
    QUEEN("QUEEN"),
    KING("KING"),
    ACE("ACE");

    private String faceValue;

    private FaceValue( String faceValue ) {
        this.faceValue = faceValue;
    }
    public String toString() {
        return this.faceValue;
    }

}

enum Suit {
    HEART("HEART"),
    SPADE("SPADE"),
    CLUB("CLUB"),
    DIAMOND("DIAMOND");

    private String suit;

    private Suit( String suit ) {
        this.suit = suit;
    }
    public String toString() {
        return this.suit;
    }
}
