from os import system, execl, _exit
import sys, subprocess
import constants

class Bash:
    @staticmethod
    def cat_data_file():
        subshell = str("(osascript -e "+\
                       "\'tell application \"Terminal\" to activate\' -e \'tell "+\
                       "application \"System Events\" to tell process \"Terminal\""+\
                       " to keystroke \"t\" using command down\' -e \'tell "+\
                       "application \"Terminal\" to do script \""+\
                       "cat "+constants.PYDIR+"data_file.csv\" "+\
                       "in selected tab of the front window\')")
        system(subshell)

    @staticmethod
    def cp(source_file, target_file):
        subshell = str("(cp "+source_file+" "+target_file+")")
        system(subshell)

    @staticmethod
    def fix_routes():
        SHFILE = constants.SHDIR + 'fixRoutes.sh'
        proc = subprocess.Popen(str(SHFILE), shell=True,
                                             stdout=subprocess.PIPE,
                                             stdin=subprocess.PIPE,
                                             stderr=subprocess.PIPE)
        proc.wait()

    @staticmethod
    def restart_script():
        print 'Restarting console.py  ...'
        python = sys.executable
        execl(python, python, * sys.argv)

    @staticmethod
    def save_to_ui(exploit_state, script_state):
        SHFILE = 'sh/sedChanges.sh'
        PYTARG = 'modules/ui.py'
        subshell = str("("+constants.PYDIR+SHFILE+" "+constants.PYDIR+PYTARG+\
                   " "+exploit_state+" "+"[\'"+'\',\''.join(map(str, script_state))+"\']"+")")
        system(subshell)
